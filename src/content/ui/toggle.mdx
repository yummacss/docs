## React

### Basic

<Preview id="toggle-basic" />

```tsx
"use client";

import { Toggle } from "@base-ui/react/toggle";
import { Heart } from "@phosphor-icons/react";
import * as React from "react";

export default function ExampleToggle() {
  const [pressed, setPressed] = React.useState(false);

  return (
    <div className="d-f g-px br-1 bw-1 bc-silver-4 bg-silver-1 p-px">
      <Toggle
        aria-label="Favorite"
        pressed={pressed}
        onPressedChange={setPressed}
        className="d-f d-8 ai-c jc-c br-1 c-slate-6 us-none h:bg-silver-2 a:bg-silver-3"
      >
        <Heart size={20} weight={pressed ? "fill" : "regular"} />
      </Toggle>
    </div>
  );
}
```
