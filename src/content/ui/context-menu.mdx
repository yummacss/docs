export const meta = {
  title: "Context Menu",
  description: "Display contextual actions on right-click.",
};

## Installation

```bash
pnpm add @base-ui/react motion
```

## Examples

Build this component with Yumma CSS & Base UI.

### Base

<Tabs defaultValue="preview">
  <TabsList>
    <TabsTab value="preview">Preview</TabsTab>
    <TabsTab value="code">Code</TabsTab>
  </TabsList>
  <TabsPanel className="bg-white bbr-sm" value="preview">
    <Preview className="h-100 bw-0" registryId="context-menu-base" />
  </TabsPanel>
  <TabsPanel value="code">```tsx registryId="context-menu-base" ```</TabsPanel>
</Tabs>
