## React

An interactive component powered by Base UI and styled with Yumma CSS.

### Basic

<Preview id="context-menu-basic" />

```tsx
import { ContextMenu } from "@base-ui/react/context-menu";
import { ContextMenu } from "@base-ui/react/context-menu";

export default function ExampleContextMenu() {
  return (
    <ContextMenu.Root>
      <ContextMenu.Trigger className="d-f ai-c jc-c h-48 w-60 br-1 bw-1 bc-silver-4 c-slate-12 us-none">
        Right click here
      </ContextMenu.Trigger>
      <ContextMenu.Portal>
        <ContextMenu.Positioner className="ow-0">
          <ContextMenu.Popup className="br-1 bg-white c-slate-12 py-1 bsh-md bw-1 bc-silver-4">
            <ContextMenu.Item className="d-f py-2 pr-8 pl-4 fs-sm us-none c-d ow-0">
              Add to Library
            </ContextMenu.Item>

            <ContextMenu.Item className="d-f py-2 pr-8 pl-4 fs-sm us-none c-d ow-0">
              Add to Playlist
            </ContextMenu.Item>

            <ContextMenu.Separator className="mx-4 my-1 h-px bg-silver-4" />
            <ContextMenu.Item className="d-f py-2 pr-8 pl-4 fs-sm us-none c-d ow-0">
              Play Next
            </ContextMenu.Item>
            <ContextMenu.Item className="d-f py-2 pr-8 pl-4 fs-sm us-none c-d ow-0">
              Play Last
            </ContextMenu.Item>
            <ContextMenu.Separator className="mx-4 my-1 h-px bg-silver-4" />
            <ContextMenu.Item className="d-f py-2 pr-8 pl-4 fs-sm us-none c-d ow-0">
              Favorite
            </ContextMenu.Item>
            <ContextMenu.Item className="d-f py-2 pr-8 pl-4 fs-sm us-none c-d ow-0">
              Share
            </ContextMenu.Item>
          </ContextMenu.Popup>
        </ContextMenu.Positioner>
      </ContextMenu.Portal>
    </ContextMenu.Root>
  );
}
```
