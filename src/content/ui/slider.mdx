## React

An interactive component powered by Base UI and styled with Yumma CSS.

### Basic

<Preview id="slider-basic" />

```tsx
import { Slider } from "@base-ui/react/slider";

export default function ExampleSlider() {
  return (
    <Slider.Root defaultValue={25}>
      <Slider.Control className="d-f ai-c w-56 py-3 us-none">
        <Slider.Track className="h-1 w-full bg-silver-2">
          <Slider.Indicator className="bg-slate-6" />
          <Slider.Thumb className="d-4 br-full bg-white bw-1 bc-silver-4 us-none" />
        </Slider.Track>
      </Slider.Control>
    </Slider.Root>
  );
}
```

## HTML

A vanilla HTML and CSS implementation.

### Basic Example

```html
<div class="w-56 py-3">
  <div class="p-r h-1 w-full bg-silver-2" style="border-radius: 0.25rem;">

    <!-- Thumb -->
    <div class="p-a t-50 d-4 br-full bg-white bw-1 bc-silver-4" style="left: 25%; transform: translate(-50%, -50%);"></div>
  </div>
```
