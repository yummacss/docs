## React

An interactive component powered by Base UI and styled with Yumma CSS.

### Basic

<Preview id="combobox-basic" />

```tsx
"use client";

import { Combobox } from "@base-ui/react/combobox";
import { CaretDownIcon, CheckIcon, XIcon } from "@phosphor-icons/react";
import { useId } from "react";

export default function ExampleCombobox() {
  const id = useId();

  return (
    <Combobox.Root items={fruits}>
      <div className="d-f fd-c g-1 fs-sm fw-500 c-slate-12 p-r">
        <label htmlFor={id}>Choose a fruit</label>
        <div className="p-r">
          <Combobox.Input
            placeholder="e.g. Apple"
            id={id}
            className="h-10 w-64 bw-1 bc-silver-4 br-1 bg-white pl-4 fs-sm c-slate-12 f:os-s f:ow-2 f:oo--1 f:oc-blue-8"
          />
          <div className="p-a d-f ai-c jc-c r-2 b-0 h-10 c-silver-9">
            <Combobox.Clear
              className="d-f ai-c jc-c d-6 h-10 br-1 bg-transparent p-0"
              aria-label="Clear selection"
            >
              <XIcon size={16} />
            </Combobox.Clear>
            <Combobox.Trigger
              className="d-f ai-c jc-c d-6 h-10 br-1 bg-transparent p-0"
              aria-label="Open popup"
            >
              <CaretDownIcon size={16} />
            </Combobox.Trigger>
          </div>
        </div>
      </div>

      <Combobox.Portal>
        <Combobox.Positioner className="ow-0" sideOffset={4}>
          <Combobox.Popup className="bw-1 bc-silver-4 br-1 bg-white bs-lg c-slate-12 o-h">
            <Combobox.Empty className="p-4 fs-sm c-silver-9">
              No fruits found.
            </Combobox.Empty>
            <Combobox.List className="o-y-auto ow-0 py-2">
              {(item: string) => (
                <Combobox.Item
                  key={item}
                  value={item}
                  className="d-g ai-c g-2 py-2 pr-8 pl-4 fs-sm ow-0 us-none c-d"
                >
                  <Combobox.ItemIndicator>
                    <CheckIcon size={12} weight="bold" />
                  </Combobox.ItemIndicator>
                  <div>{item}</div>
                </Combobox.Item>
              )}
            </Combobox.List>
          </Combobox.Popup>
        </Combobox.Positioner>
      </Combobox.Portal>
    </Combobox.Root>
  );
}

const fruits = [
  "Apple",
  "Banana",
  "Orange",
  "Pineapple",
  "Grape",
  "Mango",
  "Strawberry",
  "Blueberry",
  "Raspberry",
  "Cherry",
  "Peach",
  "Pear",
  "Kiwi",
  "Watermelon",
];
```
