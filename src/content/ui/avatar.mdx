### Circle

```tsx
import { Avatar } from "@base-ui/react/avatar";

export default function AvatarCircle() {
  return (
    <Avatar.Root className="d-if d-10 ai-c jc-c br-pill o-h bg-silver-2">
      <Avatar.Image
        src="/img/people/rayhan-zua.jpg"

  );
}
```
