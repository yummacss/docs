export const meta = {
  title: "Avatar",
  description: "Display user profile images & fallbacks.",
};

## Installation

```bash
pnpm add @base-ui/react
```

## Examples

Implement this component in your project using Yumma CSS and Base UI.

### Base

<Tabs defaultValue="preview">
  <TabsList>
    <TabsTab value="preview">Preview</TabsTab>
    <TabsTab value="code">Code</TabsTab>
  </TabsList>
  <TabsPanel className="bg-white bbr-sm" value="preview">
    <Preview className="bw-0" registryId="avatar-base" />
  </TabsPanel>
  <TabsPanel value="code">
    ```tsx registryId="avatar-base"
    ```
  </TabsPanel>
</Tabs>
