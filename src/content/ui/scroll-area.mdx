export default function ExampleScrollArea() {

    <ScrollArea.Root className="h-36 w-96 max-w-full">
      <ScrollArea.Viewport className="h-full br-1 bw-1 bc-silver-4">

            tradition, and without professional guidance. It is not a
            particular architectural movement or style, but rather a broad
            category, encompassing a wide range and variety of building types.
          </p>
          <p className="m-0">

            constrained by the materials available in its particular region and
            reflects local traditions and cultural practices.

  );
