## React

An interactive component powered by Base UI and styled with Yumma CSS.

### Basic

<Preview id="toolbar-basic" />

```tsx
import { Toolbar } from "@base-ui/react/toolbar";
import { Toggle } from "@base-ui/react/toggle";
import { ToggleGroup } from "@base-ui/react/toggle-group";
import {
  TextAlignLeft,
  TextAlignCenter,
  TextAlignRight,
  TextBolder,
  TextItalic,
  TextUnderline,
} from "@phosphor-icons/react";

export default function ExampleToolbar() {
  return (
    <Toolbar.Root className="d-f ai-c g-1 br-1 bw-1 bc-silver-4 bg-silver-1 p-1">
      <ToggleGroup className="d-f g-1" aria-label="Text alignment">
        <Toolbar.Button
          render={<Toggle />}
          aria-label="Align left"
          value="align-left"
          className="d-f d-8 ai-c jc-c br-1 c-slate-6 us-none h:bg-silver-2"
        >
          <TextAlignLeft size={16} />
        </Toolbar.Button>
        <Toolbar.Button
          render={<Toggle />}
          aria-label="Align center"
          value="align-center"
          className="d-f d-8 ai-c jc-c br-1 c-slate-6 us-none h:bg-silver-2"
        >
          <TextAlignCenter size={16} />
        </Toolbar.Button>
        <Toolbar.Button
          render={<Toggle />}
          aria-label="Align right"
          value="align-right"
          className="d-f d-8 ai-c jc-c br-1 c-slate-6 us-none h:bg-silver-2"
        >
          <TextAlignRight size={16} />
        </Toolbar.Button>
      </ToggleGroup>

      <Toolbar.Separator className="w-px h-4 mx-1 bg-silver-4" />

      <ToggleGroup className="d-f g-1" aria-label="Text formatting" multiple>
        <Toolbar.Button
          render={<Toggle />}
          aria-label="Bold"
          value="bold"
          className="d-f d-8 ai-c jc-c br-1 c-slate-6 us-none h:bg-silver-2"
        >
          <TextBolder size={16} />
        </Toolbar.Button>
        <Toolbar.Button
          render={<Toggle />}
          aria-label="Italic"
          value="italic"
          className="d-f d-8 ai-c jc-c br-1 c-slate-6 us-none h:bg-silver-2"
        >
          <TextItalic size={16} />
        </Toolbar.Button>
        <Toolbar.Button
          render={<Toggle />}
          aria-label="Underline"
          value="underline"
          className="d-f d-8 ai-c jc-c br-1 c-slate-6 us-none h:bg-silver-2"
        >
          <TextUnderline size={16} />
        </Toolbar.Button>
      </ToggleGroup>
    </Toolbar.Root>
  );
}
```
