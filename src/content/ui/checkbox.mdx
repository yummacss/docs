## React

An interactive component powered by Base UI and styled with Yumma CSS.

### Basic

<Preview id="checkbox-basic" />

```tsx
import { Checkbox } from "@base-ui/react/checkbox";
import { CheckIcon } from "@phosphor-icons/react";

export default function ExampleCheckbox() {
  return (
    <label className="d-f ai-c g-2 fs-md c-slate-12">
      <Checkbox.Root
        defaultChecked
        className="d-f ai-c jc-c d-5 br-1 fv:os-s fv:ow-2 fv:oo-2 fv:oc-blue-8 checkbox-root"
      >
        <Checkbox.Indicator className="d-f c-silver-1">
          <CheckIcon size={12} weight="bold" />
        </Checkbox.Indicator>
      </Checkbox.Root>
      Enable notifications
    </label>
  );
}
```
