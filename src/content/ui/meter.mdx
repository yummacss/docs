## React

An interactive component powered by Base UI and styled with Yumma CSS.

### Basic

<Preview id="meter-basic" />

```tsx
import { Meter } from "@base-ui/react/meter";

export default function ExampleMeter() {
  return (
    <Meter.Root className="d-g gtc-2 rg-2 w-48" value={24}>
      <Meter.Label className="fs-sm fw-500 c-slate">Storage Used</Meter.Label>
      <Meter.Value className="m-0 fs-sm c-slate ta-r gcs-2" />
      <Meter.Track className="o-h bg-silver-2 h-2">
        <Meter.Indicator className="d-b bg-slate-6 h-full" />
      </Meter.Track>
    </Meter.Root>
  );
}
```
