### Basic

<Preview id="alert-dialog-basic" variant="inlineCentered" className="bg-white" />

```tsx
import { AlertDialog } from "@base-ui/react/alert-dialog";

export default function AlertDialogBasic() {
  return (
    <AlertDialog.Root>
      <AlertDialog.Trigger className="bg-red c-white px-4 py-2 br-0 fs-sm fw-500 h:bg-red-7">
        Discard draft
      </AlertDialog.Trigger>
      <AlertDialog.Portal>
        <AlertDialog.Backdrop className="p-f zi-50 t-0 l-0 r-0 b-0 bg-black/50" />
        <AlertDialog.Popup className="p-f zi-50 t-half l-half bw-1 bc-silver-4 br-0 p-6 bg-white w-96">
          <AlertDialog.Title className="ff-s fs-lg fw-600 c-black mb-2">
            Discard draft?
          </AlertDialog.Title>
          <AlertDialog.Description className="fs-sm c-slate-6 mb-6">
            You can't undo this action.
          </AlertDialog.Description>
          <div className="d-f jc-fe g-3">
            <AlertDialog.Close className="bg-white c-black bw-1 bc-silver-4 px-3 py-2 br-0 fs-sm fw-500 h:bg-silver-1">
              Cancel
            </AlertDialog.Close>
            <AlertDialog.Close className="bg-red c-white px-3 py-2 br-0 fs-sm fw-500 h:bg-red-7">
              Discard
            </AlertDialog.Close>
          </div>
        </AlertDialog.Popup>
      </AlertDialog.Portal>
    </AlertDialog.Root>
  );
}
```

## HTML

### Basic Example

```html
<button aria-haspopup="dialog" class="bg-red c-white px-4 py-2 br-0 fs-sm fw-500 h:bg-red-7">
  Discard draft
</button>

<!-- Dialog overlay -->
<div class="p-f zi-50 t-0 l-0 w-full h-full d-f ai-c jc-c bg-black/50">
  <div role="alertdialog" aria-modal="true" class="bw-1 bc-silver-4 br-0 p-6 bg-white w-96">
    <h3 class="ff-s fs-lg fw-600 c-black mb-2">Discard draft?</h3>
    <p class="fs-sm c-slate-6 mb-6">You can't undo this action.</p>
    <div class="d-f jc-fe g-3">
      <button class="bg-white c-black bw-1 bc-silver-4 px-3 py-2 br-0 fs-sm fw-500 h:bg-silver-1">Cancel</button>
      <button class="bg-red c-white px-3 py-2 br-0 fs-sm fw-500 h:bg-red-7">Discard</button>
    </div>
  </div>
</div>
```
