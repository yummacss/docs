export const meta = {
  title: "File Upload",
  description: "Upload files with drag and drop support.",
};

## Installation

```bash
pnpm add @base-ui/react motion
```

## Examples

Implement this component in your project using Yumma CSS and Base UI.

### Base

A drag and drop zone for file uploads with a dashed border and upload icon.

<Tabs defaultValue="preview">
  <TabsList>
    <TabsTab value="preview">Preview</TabsTab>
    <TabsTab value="code">Code</TabsTab>
  </TabsList>
  <TabsPanel className="bg-white bbr-sm" value="preview">
    <Preview className="h-fc bw-0" registryId="file-upload-base" />
  </TabsPanel>
  <TabsPanel value="code">
    ```tsx registryId="file-upload-base"
    ```
  </TabsPanel>
</Tabs>
