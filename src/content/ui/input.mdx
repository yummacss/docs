## React

An interactive component powered by Base UI and styled with Yumma CSS.

### Basic

<Preview id="input-basic" variant="inlineCentered" className="bg-white" />

```tsx
import { Input } from "@base-ui/react/input";

export default function ExampleInput() {
  return (
    <Input
      placeholder="Name"
      aria-label="Name"
      className="h-10 w-full max-w-64 br-1 bw-1 bc-silver-4 pl-4 fs-md c-slate bg-white f:os-s f:ow-2 f:oo--1 f:oc-blue-8"
    />
  );
}
```
