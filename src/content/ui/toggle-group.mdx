## React

An interactive component powered by Base UI and styled with Yumma CSS.

### Basic

<Preview id="toggle-group-basic" />

```tsx
import { Toggle } from "@base-ui/react/toggle";
import { ToggleGroup } from "@base-ui/react/toggle-group";
import {
  TextAlignLeft,
  TextAlignCenter,
  TextAlignRight,
} from "@phosphor-icons/react";

export default function ExampleToggleGroup() {
  return (
    <ToggleGroup
      defaultValue={["left"]}
      className="d-f g-px br-1 bw-1 bc-silver-4 bg-silver-1 p-px"
    >
      <Toggle
        aria-label="Align left"
        value="left"
        className="d-f d-8 ai-c jc-c br-1 c-slate-6 us-none h:bg-silver-2 a:bg-silver-3"
      >
        <TextAlignLeft size={16} />
      </Toggle>
      <Toggle
        aria-label="Align center"
        value="center"
        className="d-f d-8 ai-c jc-c br-1 c-slate-6 us-none h:bg-silver-2 a:bg-silver-3"
      >
        <TextAlignCenter size={16} />
      </Toggle>
      <Toggle
        aria-label="Align right"
        value="right"
        className="d-f d-8 ai-c jc-c br-1 c-slate-6 us-none h:bg-silver-2 a:bg-silver-3"
      >
        <TextAlignRight size={16} />
      </Toggle>
    </ToggleGroup>
  );
}
```
