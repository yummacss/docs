## React

An interactive component powered by Base UI and styled with Yumma CSS.

### Basic

<Preview id="checkbox-group-basic" />

```tsx
"use client";

import { Checkbox } from "@base-ui/react/checkbox";
import { CheckboxGroup } from "@base-ui/react/checkbox-group";
import { CheckIcon } from "@phosphor-icons/react";
import { useId } from "react";

export default function ExampleCheckboxGroup() {
  const id = useId();

  return (
    <CheckboxGroup
      aria-labelledby={id}
      defaultValue={["fuji-apple"]}
      className="d-f fd-c ai-fs g-1 c-slate-12"
    >
      <div className="fw-500" id={id}>
        Apples
      </div>

      <label className="d-f ai-c g-2">
        <Checkbox.Root
          name="apple"
          value="fuji-apple"
          className="d-f ai-c jc-c d-5 br-1 fv:os-s fv:ow-2 fv:oo-2 fv:oc-blue-8 checkbox-group-root"
        >
          <Checkbox.Indicator className="d-f c-silver-1">
            <CheckIcon size={12} weight="bold" />
          </Checkbox.Indicator>
        </Checkbox.Root>
        Fuji
      </label>

      <label className="d-f ai-c g-2">
        <Checkbox.Root
          name="apple"
          value="gala-apple"
          className="d-f ai-c jc-c d-5 br-1 fv:os-s fv:ow-2 fv:oo-2 fv:oc-blue-8 checkbox-group-root"
        >
          <Checkbox.Indicator className="d-f c-silver-1">
            <CheckIcon size={12} weight="bold" />
          </Checkbox.Indicator>
        </Checkbox.Root>
        Gala
      </label>

      <label className="d-f ai-c g-2">
        <Checkbox.Root
          name="apple"
          value="granny-smith-apple"
          className="d-f ai-c jc-c d-5 br-1 fv:os-s fv:ow-2 fv:oo-2 fv:oc-blue-8 checkbox-group-root"
        >
          <Checkbox.Indicator className="d-f c-silver-1">
            <CheckIcon size={12} weight="bold" />
          </Checkbox.Indicator>
        </Checkbox.Root>
        Granny Smith
      </label>
    </CheckboxGroup>
  );
}
```
