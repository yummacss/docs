## Usage

### Grid Column Span

Allow a grid item to span a specific number of columns.

<Reference category="grid" name="grid-column" />

<Preview variant="inline">
    <div className="d-g g-4 gtc-3 ta-c">
      <div className="bg-indigo gc-s-2 p-4 c-white">A</div>
      <div className="bg-indigo-8 p-4 c-indigo-5">B</div>
      <div className="bg-indigo-8 p-4 c-indigo-5">C</div>
      <div className="bg-indigo gc-s-2 p-4 c-white">D</div>
    </div>
</Preview>

```html "gc-s-2"
<div class="d-g g-4 gtc-3 ta-c">
  <div class="bg-indigo gc-s-2 p-4 c-white">A</div>
  <div class="bg-indigo-8 p-4 c-indigo-5">B</div>
  <div class="bg-indigo-8 p-4 c-indigo-5">C</div>
  <div class="bg-indigo gc-s-2 p-4 c-white">D</div>
</div>
```
