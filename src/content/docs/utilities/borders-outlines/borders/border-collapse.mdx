import { collapse, separate } from "@constants/border-collapse";

Controls the collapsing or separation of table borders.

<Utility
  data={[
    {
      className: "bc-c",
      properties: ["border-collapse: collapse;"],
    },
    {
      className: "bc-s",
      properties: ["border-collapse: separate;"],
    },
  ]}
/>

### Collapse

Each table cell will have its own border.

<Live
  data={collapse}
  useCenter
  useHeight="200px"
  useMarks={["bc-c"]}
  usePadding
  useVersion="2.1.0"
/>

### Separate

The table borders will be merged.

<Live
  data={separate}
  useCenter
  useHeight="200px"
  useMarks={["bc-s"]}
  usePadding
  useVersion="2.1.0"
/>
