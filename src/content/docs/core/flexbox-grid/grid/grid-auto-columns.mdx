<Utility
  data={[
    {
      className: "gac-min",
      properties: ["grid-auto-columns: min-content;"],
    },
    {
      className: "gac-max",
      properties: ["grid-auto-columns: max-content;"],
    },
    {
      className: "gac-auto",
      properties: ["grid-auto-columns: auto;"],
    },
  ]}
/>
