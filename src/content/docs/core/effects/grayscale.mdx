<Class name="/filter" />

This example creates a **grayscale** effect on elements.

```html live "f-g-0" "f-g-50" "f-g-100"
<div class="d-g g-16 gtc-1 sm:gtc-3">
  <div class="dim-32 p-r rad-1">
    <img class="dim-full f-g-0 of-c rad-1" src="https://picsum.photos/id/875/300/300" />
  </div>
  <div class="dim-32 p-r rad-1">
    <img class="dim-full f-g-50 of-c rad-1" src="https://picsum.photos/id/875/300/300" />
  </div>
  <div class="dim-32 p-r rad-1">
    <img class="dim-full f-g-100 of-c rad-1" src="https://picsum.photos/id/875/300/300" />
  </div>
</div>
```

<Modifier
  utility="f-g-0"
  prefixUtil="f-g"
  modifiers={{

    responsiveUtil: "f-g-100",
  }}
