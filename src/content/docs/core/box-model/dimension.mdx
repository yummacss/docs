Controls the height and width of an element at once.

  additionalClasses={[
    { name: "1/1", value: "100dvh" },

    { name: "auto", value: "auto" },
    { name: "fc", value: "fit-content" },
    { name: "full", value: "100%" },
    { name: "half", value: "50%" },

  ]}
  classPrefix="dim-"
  incrementValue={0.25}
  propNames={["height", "width"]}
  range={100}
  unit="rem"
/>
