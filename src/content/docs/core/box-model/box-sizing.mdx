import { borderBox, contentBox } from "@consts/box-sizing";

Controls the calculation of the total size of an element.

<Utility
  data={[
    {
      className: "bs-bb",
      properties: ["box-sizing: border-box;"],
    },
    {
      className: "bs-cb",
      properties: ["box-sizing: content-box;"],
    },
  ]}
/>

<Aside type="note">
  [Stylecent](/docs/stylecent) uses the `border-box` property by default for
  elements, which is what the base styles are built on.
</Aside>

## Content Box

The height and width of an element only apply to its content.

<Preview
  data={contentBox}
  useCenter
  useHeight="200px"
  useMarks={["bs-cb"]}
  usePadding
  useVersion="2.1.0"
/>

## Border Box

The height and width of the element are the same for all parts.

<Preview
  data={borderBox}
  useCenter
  useHeight="200px"
  useMarks={["bs-bb"]}
  usePadding
  useVersion="2.1.0"
/>
