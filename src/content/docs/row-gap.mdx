## Usage

Define the spacing between grid rows.

<Preview variant="inline">
  <div id="area">
    <div className="d-g rg-10 gtr-3 c-white">
      <div className="p-4 bg-indigo ta-c">A</div>
      <div className="p-4 bg-indigo ta-c">B</div>
      <div className="p-4 bg-indigo ta-c">C</div>
    </div>
  </div>
</Preview>

```html "rg-10"
<div class="d-g rg-10 gtr-3 c-white">
  <div class="p-4 bg-indigo ta-c">A</div>
  <div class="p-4 bg-indigo ta-c">B</div>
  <div class="p-4 bg-indigo ta-c">C</div>
</div>
```
